' This example uses the optional LET statement to assign values to
' variables.
 
' To try this example:
' 1. Choose New Project from the File menu
' 2. Copy the example code below to the code window
' 3. Press F5 to run the example
 
 CLS                            ' Clear the screen
 LET D = 12
 LET E = 12 - 2
 LET F = 12 - 4
 LET SUM = D + E + F
 PRINT D; E; F; SUM
 
 D = 12                         ' The program lines to the left perform the
 E = 12 - 2                     ' same function without using the LET keyword
 F = 12 - 4
 SUM = D + E + F
 PRINT D; E; F; SUM
