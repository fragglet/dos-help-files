' The example uses the PRINTFORM method to print the current form.
 
' To try this example:
' 1. Choose New Project from the File menu
' 2. Choose New Form from the File menu to create a form
' 3. Press Alt+F4 to return to the programming environment
' 4. Copy the code example below to the form module
' 5. Press F5 to run the example
 
 DECLARE SUB PrintFormDemo ()
 
 SUB Form_Click ()
     CALL PrintFormDemo
 END SUB
 
 SUB PrintFormDemo ()
     ON LOCAL ERROR GOTO ErrorHandler ' Set up error handler
     PRINTFORM                        ' Print form
     PRINTER.ENDDOC
     EXIT SUB
 
ErrorHandler:
     Msg$ = "The form could not be printed. "
     MSGBOX Msg$                      ' Display message
     RESUME NEXT
 
 END SUB
