' This example displays a message in a label when a file name in a file list
' box is double-clicked.
 
' To try this example:
' 1. Choose New Project from the File menu
' 2. Choose New Form from the File menu to create a form with:
'    • File list box
'    • Label (sized about 3" wide)
'    • Directory list box
' 3. Press Alt+F4 to return to the programming environment
' 4. Copy the code example below to the form module
' 5. Press F5 to run the example
 
 SUB Dir1_Change ()
     File1.Path = Dir1.Path          ' When Dir1 path changes, set File1 path
 END SUB
 
 SUB File1_DblClick ()               ' Display the selected file name
    Label1.Caption = "Your selection: " + UCASE$(File1.FileName)
 END SUB
 
 SUB File1_PathChange ()
    Dir1.Path = File1.Path           ' When File1 path changes, set Dir1 path
 END SUB
