' This example illustrates the effect of changing the value of the
' TITLEBAR_BACKCOLOR and TITLEBAR_FORECOLOR constants.
 
' To try this example:
' 1. Choose New Project from the File menu
' 2. Choose New Form from the File menu to create a form with a command
'    button
' 3. Set the command button Caption property to "&DoIt"
' 4. Press Alt+F4 to return to the programming environment
' 5. Copy the code example below to the form module
' 6. Press F5 to run the example
 
 ' $INCLUDE: 'CONSTANT.BI'          ' Include ControlPanel definitions
 
 SUB Command1_Click ()
     SCREEN.ControlPanel(TITLEBAR_BACKCOLOR) = 2
     SCREEN.ControlPanel(TITLEBAR_FORECOLOR) = 4
 END SUB
