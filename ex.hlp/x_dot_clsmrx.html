' The example uses the CLS method to erase printed information on a form.
'
' To try this example:
' 1. Choose New Project from the File menu
' 2. Choose New Form from the File menu to create a form
' 3. Press Alt+F4 to return to the programming environment
' 4. Copy the code example below to the form module
' 5. Press F5 to run the example
 
 DECLARE SUB ClsDemo ()
 
 SUB ClsDemo ()
     AutoRedraw = -1                  ' Turn on AutoRedraw
     ForeColor = 15                   ' Set foreground to white
     BackColor = 1                    ' Set background to blue
     Msg$ = "This is information printed on the form background."
     PRINT Msg$                       ' Print message to form
     Msg$ = "Choose OK to clear the information"
     Msg$ = Msg$ + " just displayed on the form"
     MSGBOX Msg$                      ' Display message
     CLS                              ' Clear form background
 END SUB
 
 SUB Form_Click ()
         CALL ClsDemo
 END SUB
