' This example uses READ to read in values from a DATA statement.
' RESTORE allows the data to be reread.
 
' To try this example:
' 1. Choose New Project from the File menu
' 2. Copy the code example below to the code window
' 3. Press F5 to run the example
 
 CLS                 ' Clear the screen
 FOR i% = 1 TO 3
     READ a%, b$
     PRINT a%, b$
     RESTORE         ' Restore so data can be read again
 NEXT i%
 PRINT
 FOR i% = 1 TO 3
     READ a%, b$
     PRINT a%, b$
 NEXT i%
 
 DATA 1, "first", 2, "second", 3, "third"
