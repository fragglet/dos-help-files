' This example illustrates the effect of changing the value of the
' ACTIVE_WINDOW_SHADOW constant.
 
' To try this example:
' 1. Choose New Project from the File menu
' 2. Choose New Form from the File menu to create a form with a check box
' 3. Set the check box Caption property to "Shadow Toggle"
' 4. Press Alt+F4 to return to the programming environment
' 5. Copy the code example below to the form module
' 6. Press F5 to run the example
 
 ' $INCLUDE: 'CONSTANT.BI'          ' Include ControlPanel definitions
 
 SUB Check1_Click ()
     IF SCREEN.ControlPanel(ACTIVE_WINDOW_SHADOW) = -1 THEN
          SCREEN.ControlPanel(ACTIVE_WINDOW_SHADOW) = 0
     ELSE
          SCREEN.ControlPanel(ACTIVE_WINDOW_SHADOW) = -1
     END IF
 END SUB
