' This example uses the EXP and LOG functions to print the value of e and
' calculates the natural logarithm of e taken to the first, second, and
' third powers.
 
 CLS                              ' Clear the screen
 PRINT EXP(1),
 FOR I = 1 TO 3
     PRINT LOG(EXP(1) ^ I),
 NEXT
