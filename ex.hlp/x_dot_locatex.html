' This example illustrates one use of the CSRLIN and POS function, and the
' LOCATE statement.
 
' To try this example:
' 1. Choose New Form from the File menu to create a form
' 2. Set the form's AutoRedraw property to True.
' 3. Press Alt+F4 to return to the programming environment
' 4. Copy the code example below to the form module
' 5. Press F5 to run the example
 
 SUB Form_Click()
     SCREEN.HIDE                           ' Hide any visible forms
     LOCATE Form1.Top + 1, Form1.Left + 1  ' Move cursor
     col% = CSRLIN                         ' Get cursor column position
     row% = POS(1)                         ' Get cursor row position
     PRINT row%, col%
     SCREEN.SHOW                           ' Redisplay and hidden forms
 END SUB
