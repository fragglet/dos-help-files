' This example demonstrates the MouseDown and MouseUp events. When you run
' the example, click the form to display messages generated in the mouse
' events procedures above.
 
' To try this example:
' 1. Choose New Project from the File menu
' 2. Choose New Form from the File menu to create a form
' 3. Press Alt+F4 to return to the programming environment
' 4. Copy the code example below to the form module
' 5. Press F5 to run the example
 
 SUB Form_Load ()
     ForeColor = 8
 END SUB
 
 SUB Form_MouseDown (Button AS INTEGER, Shift AS INTEGER, X AS _
 SINGLE, Y AS SINGLE)
    PRINT "The MouseDown event occurred"
 END SUB
 
 SUB Form_MouseUp (Button AS INTEGER, Shift as Integer, X AS _
 SINGLE, Y AS SINGLE)
     PRINT (CHR$(13) + CHR$(10) + "The MouseUp event occurred")
 END SUB
