  Include:   <string.h>
 
  Syntax:    char *strncat( char *string1, char *string2,
                            size_t count );
 
             char _far *_fstrncat( char _far *string1, char _far *string2,
                                   size_t count );
 
  Returns:   a pointer to the concatenated string.
