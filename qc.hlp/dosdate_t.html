  Include:  <dos.h>
 
  Structure:
 
  struct dosdate_t
  {                             // Current date structure
      unsigned char day;        // Day of the month; 1-31
      unsigned char month;      // Month of the year; 1-12
      unsigned int year;        // Year; 1980 - 2099
      unsigned char dayofweek;  // Day of the week, 0-6; Sunday = 0
  };
