/* KBHIT.C illustrates:
 *      kbhit
 */
 
#include <conio.h>
 
void main()
{
    /* Display message until key is pressed. */
    while( !kbhit() )
        cputs( "Hit me!! " );
 
    /* Use getch to throw key away. */
    getch();
}
