────────────────────────────────────────────────────────────────────────────
 
  Include:   <malloc.h>
 
  Syntax:    void *_expand( void *memblock, size_t size );
 
             void _based( void ) *_bexpand( _segment seg,
                                            void _based( void ) *memblock,
                                            size_t size );
 
             void _far *_fexpand( void _far *memblock, size_t size );
 
             void _near *_nexpand( void _near *memblock, size_t size );
 
  Returns:   a pointer to the reallocated memory block if successful, or
             NULL if not (-1 if _bexpand fails).
 
  See also:  calloc, free, malloc, _msize, realloc
