  struct dostime_t
  {                          // Current system time
      unsigned char hour;    // Hour; 0-23
      unsigned char minute;  // Minute; 0-59
      unsigned char second;  // Second; 0-59
      unsigned char hsecond; // 1/100 second; 0-99
  };
