────────────────────────────────────────────────────────────────────────────
 
  Include:   <malloc.h>, <stdlib.h>
 
  Syntax:    void *realloc( void *memblock, size_t size );
 
             void _based( void ) *_brealloc( _segment seg,
                          void _based( void ) *memblock,
                          size_t size );
 
             void _far *_frealloc( void _far *memblock,
                                   size_t size );
 
             void _near *_nrealloc( void _near *memblock,
                                    size_t size );
 
  Returns:   a pointer to the reallocated memory if successful, or
             NULL if not.
 
  See also:  calloc, free, malloc
