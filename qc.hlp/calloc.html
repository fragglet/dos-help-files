  Include:   <malloc.h>, <stdlib.h>
 
  Syntax:    void *calloc( size_t num, size_t size );
 
             void _based( void ) *_bcalloc( _segment seg, size_t num,
                                            size_t size );
 
             void _far *_fcalloc( size_t num, size_t size );
 
             void _near *_ncalloc( size_t num, size_t size );
 
  Returns:   a pointer to the allocated space if successful,
             or NULL if not (-1 if _bcalloc fails).
 
  See also:  free, halloc, hfree, malloc, realloc
