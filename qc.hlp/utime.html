  Include:   <sys\types.h>, <sys\utime.h>, <errno.h>
 
  Syntax:    int utime( char *filename, struct utimbuf *times );
 
  Returns:   0 if successful, or -1 if not.
             errno: EACCES, EINVAL, EMFILE, ENOENT
 
  See also:  asctime, ctime, fstat, ftime, gmtime, localtime,
             stat, time
