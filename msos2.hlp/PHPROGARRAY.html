#define INCL_WINPROGRAMLIST
 
typedef struct _HPROGARRAY {    /* hpga */
    HPROGRAM ahprog[1];
} HPROGARRAY;
 
The HPROGARRAY structure contains an array of program handles returned by
PrfQueryProgramHandle function.
 
Field      Description
────────────────────────────────────────────────────────────────────────────
 
ahprog[1]  Identifies the program handle.
