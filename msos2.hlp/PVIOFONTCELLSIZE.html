#define INCL_VIO
 
typedef struct _VIOFONTCELLSIZE {    /* viofcsz */
    LONG cx;
    LONG cy;
} VIOFONTCELLSIZE;
 
The VIOFONTCELLSIZE structure specifies the size of a font cell.
 
Field  Description
────────────────────────────────────────────────────────────────────────────
 
cx     Specifies the width of the font cell.
 
cy     Specifies the length of the font cell.
