PROGRAM byref;
 
{ BYREF.PAS : Demonstrates passing by reference. }
 
VAR
    var1, var2 : Integer;
 
PROCEDURE swap_vars( VAR var1: Integer; VAR var2: Integer );
VAR
    temp : Integer;
BEGIN
    temp := var1;
    var1 := var2;
    var2 := temp;
END; { procedure swap_vars }
 
BEGIN
 
    var1 := 55;
    var2 := 99;
    Writeln( 'var1 = ', var1, '  var2 = ', var2 );
    swap_vars( var1, var2 );
    Writeln( 'var1 = ', var1, '  var2 = ', var2 );
 
END.
