PROGRAM funct;
 
{ FUNCT.PAS : Demonstrates function basics. }
 
VAR
    num, expo, powr : Real;
 
FUNCTION power( base, exponent : Real ) : Real;
BEGIN
    IF (base > 0) THEN
        power := Exp( exponent * Ln( base ) )
    ELSE
        power := -1.0;
END;
 
BEGIN
 
    Write( 'Enter a number: ' );
    Readln( num );
    Write( 'Enter an exponent: ' );
    Readln( expo );
    powr := power( num, expo );
    Writeln( num, ' ^ ', expo, ' = ', powr );
 
END.
