PROGRAM noise;
 
{ NOISE.PAS uses the following procedures to illustrate use of
  the speaker:
 
      Delay    Inc    NoSound    Sound
}
 
USES
    Crt;
 
CONST
    program_name = 'NOISE ';
    program_desc = 'generates sound from the computer''s speaker.';
    rest         = 1000;
 
VAR
    interval : Word;
 
BEGIN
 
    Writeln( program_name, program_desc );
    Writeln;
 
    interval := 100;
    REPEAT
        Sound( interval );
        Delay( rest );
        NoSound;
        Inc( interval, 100 );
    UNTIL interval > 1000;
 
END.
