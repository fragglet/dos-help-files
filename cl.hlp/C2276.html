     Compiler error C2276
 
     illegal address of bound member function expression
 
     The address of a virtual function was taken.
 
     The following is an example of this error:
 
          class A
          {
          public:
             virtual int foo();
          } a;
          int (*pf)() = &a.foo;   // error
 
          class B
          {
          public:
             void mf()
             {
                 &this->mf;       //error
             }
          };
