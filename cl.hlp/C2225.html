     Compiler error C2225
 
     'function' : function allocation must be in a named segment
 
     The specified function allocation was not in a named segment.
 
     The following examples cause this error:
 
          char c;
          __segment segvar = (__segment)&c;
 
          void  __based((__segment)&c)      one()   {}   // error
          void  __based(segvar)             two()   {}   // error
          void  __based((__segment)__self)  three() {}   // error
          void  __based(void)               four()  {}   // error
 
          void  __based(__segname("SEG"))   five()  {}   // OK
