     Compiler warning (level 3) C4097
 
     nonstandard extension : typedef-name 'identifier1' used as synonym
     for class-name 'identifier2'
 
         typedef struct _T
         {
            int i;
         } T;
 
         struct U : T
         {                 // warning
            int j;
         };
 
         struct V : _T
         {                 // OK
            int j;
         };
