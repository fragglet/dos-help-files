'This example shows how the WRITE statement outputs delimiters between
'data items.
 
CLS                 'Clear the screen.
A=80 : B=90 : C$="That's all." : D=-1.0E-13
WRITE A,B,C$,D
PRINT A,B,C$,D
 
'Sample Output
'
'80,90,"That's all.",-1E-13
' 80            90           That's all.  -1E-13
