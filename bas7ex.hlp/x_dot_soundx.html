'This example uses the SOUND statement to produce rising and descending
'glissando.
 
CLS
FOR I = 440 TO 1000 STEP 5
   SOUND I, I/1000
NEXT
FOR I = 1000 TO 440 STEP -5
  SOUND I, I/1000
NEXT
