'This example uses a DEF FN function to calculate the factorial of an integer
'(for example, the factorial of 3 is 3*2*1).
 
DEF FNFactorial# (X%)
   STATIC Tmp#, I%
   Tmp# = 1
   FOR I% = 2 TO X%
      Tmp# = Tmp# * I%
   NEXT I%
   FNFactorial# = Tmp#
END DEF
 
INPUT "Enter an integer: ", Num%
PRINT : PRINT Num%; "factorial is"; FNFactorial#(Num%)
