'This example uses the optional LET statement to assign values to variables.
 
LET D = 12
LET E = 12 - 2
LET F = 12 - 4
LET SUM = D + E + F
PRINT D E F SUM
 
'The following program lines perform the same function without using
'the LET keyword.
 
D = 12
E = 12 - 2
F = 12 - 4
SUM = D + E + F
PRINT D E F SUM
 
'Sample Output
'
'12 10 8 30
