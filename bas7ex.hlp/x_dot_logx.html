'This example prints the value of e and uses the LOG function
'to calculate the natural logarithm of e taken to the first,
'second, and third powers.
 
CLS                    'Clear screen.
PRINT EXP(1),
FOR I = 1 TO 3
    PRINT LOG(EXP(1) ^ I),
NEXT
 
'Sample Output
'
'2.718282       1       2       3
