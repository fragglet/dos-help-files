  Include:   <graph.h>
 
  Syntax:    short _ellipse( short control, short x1, short y1,
                             short x2, short y2 );
 
             short _ellipse_w( short control, double wx1, double wy1,
                               double wx2, double wy2 );
 
             short _ellipse_wxy( short control,
                                 struct _wxycoord _far *pwxy1,
                                 struct _wxycoord _far *pwxy2 );
 
             control: _GFILLINTERIOR, _GBORDER
 
  Returns:   a nonzero value if the ellipse is drawn successfully, or
             0 if not.
 
  See also:  _arc, _floodfill, _lineto, _pie, _rectangle, _setcolor,
             _setfillmask
