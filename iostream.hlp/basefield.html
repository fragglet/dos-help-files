  static const long basefield;
 
  Remarks
 
  A mask that can be used to obtain the current radix flag bits (dec, oct,
  or hex).
 
  Example
 
  extern ostream os;
  if( ( os.flags() & ios::basefield ) == ios::hex ) .....
