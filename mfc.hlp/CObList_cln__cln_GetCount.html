  int GetCount() const;
 
  Remarks
 
  Gets the number of elements in this list.
 
  Return Value
 
  An integer value containing the element count.
 
  Example
 
  CObList list;
 
  list.AddHead( new CAge( 21 ) );
  list.AddHead( new CAge( 40 ) ); // List now contains (40, 21)
  ASSERT( list.GetCount() == 2 );
