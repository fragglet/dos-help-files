  BOOL operator ==( CTimeSpan timeSpan ) const;
  BOOL operator !=( CTimeSpan timeSpan ) const;
  BOOL operator <( CTimeSpan timeSpan ) const;
  BOOL operator >( CTimeSpan timeSpan ) const;
  BOOL operator <=( CTimeSpan timeSpan ) const;
  BOOL operator >=( CTimeSpan timeSpan ) const;
 
  Remarks
 
  These operators compare two relative time values. They return TRUE if
  the condition is true; otherwise FALSE.
 
  Example
 
  CTimeSpan ts1( 100 );
  CTimeSpan ts2( 110 );
  ASSERT( ( ts1 != ts2 ) && ( ts1 < ts2 ) && ( ts1 <= ts2 ) );
