LEN Function Programming Example
 
This example prints the length of a string and the size in bytes of several
types of variables.
 
CLS    ' Clear screen
TYPE EmpRec
   EmpName AS STRING * 20
   EmpNum AS INTEGER
END TYPE
DIM A AS INTEGER, B AS LONG, C AS SINGLE, D AS DOUBLE
DIM E AS EmpRec
 
PRINT "A string:" LEN("A string.")
PRINT "Integer:" LEN(A)
PRINT "Long:" LEN(B)
PRINT "Single:" LEN(C)
PRINT "Double:" LEN(D)
PRINT "EmpRec:" LEN(E)
END
 
Sample Output
 
A string: 9
Integer: 2
Long: 4
Single: 4
Double: 8
EmpRec: 22
