─────Run-Time Library───────────────────────────────────────────────────────
 
  Include:   <string.h>
 
  Syntax:    char *strpbrk( char *string1, char *string2 );
 
             char __far *_fstrpbrk( char __far *string1,
                                    char _far *string2 );
 
  Returns:   a pointer to the first matching character in <string1>,
             or NULL if no match is found.
 
  See also:  strspn, strcspn, strtok, strchr, strrchr, strstr
