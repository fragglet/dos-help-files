─────Run-Time Library───────────────────────────────────────────────────────
 
  Include:   <process.h>, <stdio.h>, <errno.h>
 
  Syntax:    int _spawnv( int mode, char *cmdname, char **argv );
 
             int _spawnve( int mode, char *cmdname, char **argv,
                           char **envp );
 
             int _spawnvp( int mode, char *cmdname, char **argv );
 
             int _spawnvpe( int mode, char *cmdname, char **argv,
                            char **envp );
 
             mode: _P_OVERLAY, _P_WAIT
 
  Returns:   the child process's exit status (mode = _P_WAIT) if
             successful, or -1 if not.
             errno: E2BIG, EINVAL, ENOENT, ENOEXEC, ENOMEM
 
  See also:  _spawnl..., _exec..., system
