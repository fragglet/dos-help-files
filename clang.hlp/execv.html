─────Run-Time Library───────────────────────────────────────────────────────
 
  Include:   <process.h>, <errno.h>
 
  Syntax:    int _execv( char *cmdname, char **argv );
             int _execve( char *cmdname, char **argv, char **envp );
             int _execvp( char *cmdname, char **argv );
             int _execvpe( char *cmdname, char **argv, char **envp );
 
  Returns:   -1 if unsuccessful.
             errno: E2BIG, EACCES, EMFILE, ENOENT, ENOEXEC, ENOMEM
 
  See also:  _execl..., _spawn..., system
