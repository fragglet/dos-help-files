─────Run-Time Library───────────────────────────────────────────────────────
 
  Include:   <fcntl.h>, <io.h>, <share.h>, <sys\types.h>, <sys\stat.h>,
             <errno.h>
 
  Syntax:    int _sopen( char *filename, int oflag,
                         int shflag [, int pmode] );
             oflag:  _O_APPEND, _O_BINARY, _O_CREAT, _O_EXCL, _O_RDONLY,
                     _O_RDWR, _O_TEXT, _O_TRUNC, _O_WRONLY
                     (may be joined by |)
             shflag: _SH_COMPAT, _SH_DENYNO, _SH_DENYRD, _SH_DENYRW,
                     _SH_DENYWR
             pmode:  _S_IWRITE, _S_IREAD, _S_IREAD | _S_IWRITE
 
  Returns:   a handle if successful, or -1 if not.
             errno: EACCES, EEXIST, EINVAL, EMFILE, ENOENT
 
  See also:  _fsopen, _locking, _open, _creat, fopen
