  Include:  <sys\timeb.h>
 
  Context:  _ftime
 
  Structure:
 
  struct _timeb
  {
      time_t time;              // Time in seconds since midnight
                                //   12/31/1899
      unsigned short millitm;   // Fraction in milliseconds
      short timezone;           // Time zone relative to GMT
      short dstflag;            // Daylight savings flag
  };
