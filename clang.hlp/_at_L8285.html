/* WRAP.C illustrates:
 *      _wrapon
 */
 
#include <conio.h>
#include <graph.h>
 
void main()
{
    short i;
 
    _clearscreen( _GCLEARSCREEN );
    _settextwindow( 1, 1, 25, 38 );
    _wrapon( _GWRAPON );
    for( i = 1; i < 80; i++ )
        _outtext( "Wrap on! " );
 
    _settextwindow( 1, 41, 25, 80 );
    _wrapon( _GWRAPOFF );
    for( i = 1; i < 80; i++ )
        _outtext( "Wrap off!  " );
    _getch();
    _clearscreen( _GCLEARSCREEN );
}
