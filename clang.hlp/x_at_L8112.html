─────Run-Time Library───────────────────────────────────────────────────────
 
   The following functions control input and output of streams to
   files, buffers, and devices:
 
   clearerr      _fileno        fseek        putc          sprintf
   fclose        _flushall      fsetpos      putchar       sscanf
  _fcloseall      fopen        _fsopen       puts         _tempnam
  _fdopen         fprintf       ftell       _putw          tmpnam
   feof           fputc         fwrite       rewind        tmpfile
   ferror        _fputchar      getc        _rmtmp         ungetc
   fflush         fputs         getchar      scanf         vfprintf
   fgetc          fread         gets         setbuf        vprintf
  _fgetchar       freopen      _getw         setvbuf      _vsnprintf
   fgetpos        fscanf        printf      _snprintf      vsprintf
   fgets
