─────Run-Time Library───────────────────────────────────────────────────────
 
  Include:   <io.h>
 
  Syntax:    int _wgetexit( void );
 
  Returns:   _WINEXITPROMPT, _WINEXITNOPERSIST, or _WINEXITPERSIST
             if successful, or -1 if not.
 
  See also:  _fwopen, _wabout, _wclose, _wgetfocus, _wgetscreenbuf,
             _wgetsize, _wmenuclick, _wopen, _wsetexit, _wsetfocus,
             _wsetscreenbuf, _wsetsize, _wyield
