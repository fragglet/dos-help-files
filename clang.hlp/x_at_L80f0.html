/* SWAB.C illustrates:
 *      _swab
 */
 
#include <stdlib.h>
#include <stdio.h>
 
char from[] = "BADCFEHGJILKNMPORQTSVUXWZY";
char to[] =   "..........................";
 
void main()
{
    printf( "Before:\t%s\n\t%s\n\n", from, to );
    _swab( from, to, sizeof( from ) );
    printf( "After:\t%s\n\t%s\n\n", from, to );
}
