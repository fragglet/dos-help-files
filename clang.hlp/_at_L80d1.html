─────Run-Time Library───────────────────────────────────────────────────────
 
/* KBHIT.C illustrates:
 *      _kbhit
 */
 
#include <conio.h>
 
void main()
{
    /* Display message until key is pressed. */
    while( !_kbhit() )
        _cputs( "Hit me!! " );
 
    /* Use _getch to throw key away. */
    _getch();
}
