  Include:   <dos.h>, <errno.h>
 
  Syntax:    unsigned _dos_setblock( unsigned size, unsigned seg,
                                     unsigned *maxsize );
 
  Returns:   0 if successful, or the DOS error code if not.
             errno: ENOMEM
