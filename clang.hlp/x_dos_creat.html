─────Run-Time Library───────────────────────────────────────────────────────
 
  Include:   <dos.h>, <errno.h>
 
  Syntax:    unsigned _dos_creat( char *filename, unsigned attrib,
                                  int *handle );
 
             unsigned _dos_creatnew( char *filename, unsigned attrib,
                                     int *handle );
 
  Returns:   0 if successful, or the DOS error code if not.
             errno: EACCES, EEXIST, EMFILE, ENOENT
 
  See also:  _dos_close, _dos_open, _creat, _open
