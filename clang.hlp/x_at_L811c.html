  _bios_keybrd     _dosexterr        _dos_read         _hardresume

  _bios_printer    _dos_findnext     _dos_setdate      _inp
  _bios_serialcom  _dos_freemem      _dos_setdrive     _inpw
  _bios_timeofday  _dos_getdate      _dos_setfileattr  _int86
  _bdos            _dos_getdiskfree  _dos_setftime     _int86x
  _chain_intr      _dos_getdrive     _dos_settime      _intdos
  _disable         _dos_getfileattr  _dos_setvect      _intdosx
  _dos_allocmem    _dos_getftime     _dos_write        _outp
  _dos_close       _dos_gettime      _enable           _outpw
  _dos_commit      _dos_getvect      _FP_OFF           _segread
