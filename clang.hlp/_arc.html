─────Run-Time Library───────────────────────────────────────────────────────
 
  Include:   <graph.h>
 
  Syntax:    short _arc( short x1, short y1, short x2, short y2,
                         short x3, short y3, short x4, short y4 );
 
             short _arc_w( double x1, double y1, double x2, double y2,
                           double x3, double y3, double x4, double y4 );
 
             short _arc_wxy( struct _wxycoord __far *pwxy1,
                             struct _wxycoord __far *pwxy2,
                             struct _wxycoord __far *pwxy3,
                             struct _wxycoord __far *pwxy4 );
 
  Returns:   a nonzero value if the arc is drawn successfully, or
             0 if not.
 
  See also:  _getarcinfo, _ellipse, _pie
