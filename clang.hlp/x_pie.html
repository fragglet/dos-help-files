─────Run-Time Library───────────────────────────────────────────────────────
 
  Include:   <graph.h>
 
  Syntax:    short _pie( short control,
                         short x1, short y1, short x2, short y2,
                         short x3, short y3, short x4, short y4 );
 
             short _pie_w( short control,
                           double x1, double y1, double x2, double y2,
                           double x3, double y3, double x4, double y4 );
 
             short _pie_wxy( short control,
                             struct _wxycoord __far *pwxy1,
                             struct _wxycoord __far *pwxy2,
                             struct _wxycoord __far *pwxy3,
                             struct _wxycoord __far *pwxy4 );
 
             control: _GFILLINTERIOR, _GBORDER
 
  Returns:   a nonzero value if the pie is drawn successfully, or
             0 if not.
 
  See also:  _arc, _getarcinfo, _ellipse
